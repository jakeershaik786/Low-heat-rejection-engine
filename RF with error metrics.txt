import pandas as pd
import numpy as np
from sklearn.ensemble import RandomForestRegressor
from sklearn.metrics import r2_score, mean_absolute_error, mean_squared_error
from sklearn.model_selection import train_test_split
from sklearn.preprocessing import StandardScaler
import matplotlib.pyplot as plt

# Load the data

data = pd.read_csv('/content/expfile.csv')

# Select the relevant columns
cleaned_data = data[['Frequency','Amplitude']].dropna()

# Extract the independent and dependent variables
X = cleaned_data['Frequency'].values.reshape(-1, 1)
y = cleaned_data['Amplitude'].values

# Normalize the independent variable
scaler = StandardScaler()
X_normalized = scaler.fit_transform(X)

# Split the data into training and testing sets
X_train, X_test, y_train, y_test = train_test_split(X_normalized, y, test_size=0.2, random_state=33)

# Fit the Random Forest regression model
rf_model = RandomForestRegressor(n_estimators=60, random_state=42)
rf_model.fit(X_train, y_train)

# Predict y values using the Random Forest regression model
y_train_pred = rf_model.predict(X_train)
y_test_pred = rf_model.predict(X_test)

# Calculate metrics for the training set
r2_train = r2_score(y_train, y_train_pred)
mae_train = mean_absolute_error(y_train, y_train_pred)
mse_train = mean_squared_error(y_train, y_train_pred)
rmse_train = np.sqrt(mse_train)

# Calculate metrics for the test set
r2_test = r2_score(y_test, y_test_pred)
mae_test = mean_absolute_error(y_test, y_test_pred)
mse_test = mean_squared_error(y_test, y_test_pred)
rmse_test = np.sqrt(mse_test)

# Print the metrics
print(f'Training R² score: {r2_train}')
print(f'Training MAE: {mae_train}')
print(f'Training MSE: {mse_train}')
print(f'Training RMSE: {rmse_train}\n')

print(f'Test R² score: {r2_test}')
print(f'Test MAE: {mae_test}')
print(f'Test MSE: {mse_test}')
print(f'Test RMSE: {rmse_test}\n')

# Predict for an unknown input (normalized)
unknown_input = np.array([[27.7]])  # Example unknown input
unknown_input_normalized = scaler.transform(unknown_input)
unknown_output = rf_model.predict(unknown_input_normalized)

print(f'Prediction for unknown input {unknown_input[0][0]}: {unknown_output[0]}')


# Scatter plot of actual vs predicted values for the test set with bold text, bold points, high DPI, and small figure size
plt.figure(figsize=(6, 4), dpi=300)

plt.scatter(X_test, y_test, color='blue', s=12, label='Actual values')
plt.scatter(X_test, y_test_pred, color='red', s=12, label='Predicted values')

# Make text bold
plt.xlabel('Natural Frequency (Hz)', fontsize=12, fontweight='bold')  # Uncomment and modify as needed
plt.ylabel('Amplitude (mV)', fontsize=8, fontweight='bold')
plt.title('Actual vs Predicted Values (Test Set)', fontsize=10, fontweight='bold')
plt.legend(prop={'size': 8, 'weight': 'bold'})
plt.grid(True)

# Save the figure with high DPI
plt.savefig('actual_vs_predicted_values_test_set.png', dpi=300, bbox_inches='tight')
plt.show()


# Line plot of predictions with bold text, bold regression line, high DPI, and small figure size
plt.figure(figsize=(6, 4), dpi=300)

# Sort the indices based on X_test values
sorted_idx = np.argsort(X_test[:, 0])

# Plot the test data points and regression line
plt.scatter(X_test[:, 0], y_test, color='blue', s=12, label='Test data points')
plt.plot(X_test[sorted_idx, 0], y_test_pred[sorted_idx], color='red', linewidth=1.5, label='Random Forest regression line')

# Make text bold
#plt.xlabel('Load', fontsize=12, fontweight='bold')
plt.ylabel('Amplitude(mV)', fontsize=12, fontweight='bold')
plt.title('Random Forest Regression Analysis', fontsize=11, fontweight='bold')
plt.legend(prop={'size': 9, 'weight': 'bold'})
plt.grid(True)

# Save the figure with high DPI
plt.savefig('random_forest_regression_analysis.png', dpi=300, bbox_inches='tight')
plt.show()
