# Polynomial regression
import pandas as pd
import numpy as np
from sklearn.preprocessing import PolynomialFeatures, StandardScaler
from sklearn.linear_model import LinearRegression
from sklearn.metrics import r2_score, mean_absolute_error, mean_squared_error
from sklearn.model_selection import train_test_split
import matplotlib.pyplot as plt

# Load the data

data = pd.read_csv('/content/expfile.csv')

# Select the relevant columns
cleaned_data = data[['Frequency','Amplitude']].dropna()

# Extract the independent and dependent variables
X = cleaned_data['Frequency'].values.reshape(-1, 1)
y = cleaned_data['Amplitude'].values

# Normalize the independent variable
scaler = StandardScaler()
X_normalized = scaler.fit_transform(X)

# Create polynomial features
poly = PolynomialFeatures(degree=2)  # You can change the degree to fit your needs
X_poly = poly.fit_transform(X_normalized)

# Split the data into training and testing sets
X_train, X_test, y_train, y_test = train_test_split(X_poly, y, test_size=0.2, random_state=42)

# Fit the Polynomial regression model
poly_model = LinearRegression()
poly_model.fit(X_train, y_train)

# Predict y values using the Polynomial regression model
y_train_pred = poly_model.predict(X_train)
y_test_pred = poly_model.predict(X_test)

# Calculate metrics for the training set
r2_train = r2_score(y_train, y_train_pred)
mae_train = mean_absolute_error(y_train, y_train_pred)
mse_train = mean_squared_error(y_train, y_train_pred)
rmse_train = np.sqrt(mse_train)

# Calculate metrics for the test set
r2_test = r2_score(y_test, y_test_pred)
mae_test = mean_absolute_error(y_test, y_test_pred)
mse_test = mean_squared_error(y_test, y_test_pred)
rmse_test = np.sqrt(mse_test)

# Print the metrics
print(f'Training R² score: {r2_train}')
print(f'Training MAE: {mae_train}')
print(f'Training MSE: {mse_train}')
print(f'Training RMSE: {rmse_train}\n')

print(f'Test R² score: {r2_test}')
print(f'Test MAE: {mae_test}')
print(f'Test MSE: {mse_test}')
print(f'Test RMSE: {rmse_test}\n')

# Predict for an unknown input (normalized)
unknown_input = np.array([[352.29]])  # Example unknown input
unknown_input_normalized = scaler.transform(unknown_input)
unknown_input_poly = poly.transform(unknown_input_normalized)
unknown_output = poly_model.predict(unknown_input_poly)

print(f'Prediction for unknown input {unknown_input[0][0]}: {unknown_output[0]}')

figure_width = 6  # in inches

# Scatter plot of actual vs predicted values for the test set with bold text, bold points, high DPI, and smaller figure size
plt.figure(figsize=(figure_width, figure_height), dpi=300)
plt.scatter(X_test[:, 1], y_test, color='blue', s=15, label='Actual values')
plt.scatter(X_test[:, 1], y_test_pred, color='red', s=15, label='Predicted values')
plt.xlabel('Natural Frequency (Hz)', fontsize=11, fontweight='bold')
plt.ylabel('Amplitude (mV)', fontsize=11, fontweight='bold')
plt.title('Actual vs Predicted Values (Test Set)', fontsize=14, fontweight='bold')
plt.legend(prop={'size': 9, 'weight': 'bold'})
plt.grid(True)

plt.savefig('actual_vs_predicted_values_test_set.png', dpi=300, bbox_inches='tight')
